/*global angular, $, _ */
 
angular.module('voyager.queue')
    .config(['localStorageServiceProvider', function(localStorageServiceProvider) {
        'use strict';
        localStorageServiceProvider.setPrefix('vs');
    }])
    .factory('queueService', function(localStorageService, queryBuilder, $http, $q) {
        'use strict';
        var observers = [],
            QUEUE_STORAGE_NAME = 'cart-item';
 
        var _notify = function (itemMap) {
            var length = _.size(itemMap);
            observers.forEach(function (entry) {
                entry(length, itemMap);
            });
        };
 
        var _getItems = function () {
            var items = localStorageService.get(QUEUE_STORAGE_NAME);
            if (items) {
                return items;
            }
            return {};
        };
 
        function _setItems(items) {
            var itemMap = {};
            $.each(items, function( index, value ) {
                itemMap[value.id] = value.id;
            });
            localStorageService.add(QUEUE_STORAGE_NAME, itemMap);
            _notify(itemMap);
        }
 
        return {
            addItem: function (item) {
                var itemMap = _getItems();
 
                itemMap[item.id] = {id: item.id, title: item.title};
                localStorageService.add(QUEUE_STORAGE_NAME, itemMap);
                _notify(itemMap);
            },
            addItems: function (items) {
                var itemMap = _getItems();
                $.each(items, function( index, item ) {
                    itemMap[item.id] = {id: item.id, title: item.title};
                });
                localStorageService.add(QUEUE_STORAGE_NAME, itemMap);
                _notify(itemMap);
            },
            getItems: function () {
                return _getItems();
            },
            setItems: function(items) {
                _setItems(items);
            },
            getCount: function () {
                return _.size(_getItems());
            },
            addObserver: function (observer) {
                observers.push(observer);
            },
            remove: function (id) {
                var itemMap = _getItems();
                delete itemMap[id];
                localStorageService.add(QUEUE_STORAGE_NAME, itemMap);
                _notify(itemMap);
            },
            isInCart: function (id) {
                return angular.isDefined(_getItems()[id]);
            },
            getItemIds: function() {
                return $.map(_getItems(), function (item) {
                    return item;
                });
            },
            fetch:function() {
                var deferred = $q.defer();
                var query = queryBuilder.doByIds(this.getItemIds());
 
                $http.jsonp(query).success(function (data) {
                    var response = {'docs': data.response.docs, 'bbox':data['extent.bbox']};
                    _setItems(data.response.docs);
                    deferred.resolve(response);
                });
                return deferred.promise;
            },
            clear: function() {
                localStorageService.remove(QUEUE_STORAGE_NAME);
                _notify({});
            },
            getItemsArray: function() {
                var itemsArray = [];
 
                $.map(_getItems(), function(item){
                    itemsArray.push(item);
                });
 
                return itemsArray;
            }
        };
 
    });
